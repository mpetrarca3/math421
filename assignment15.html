# Creating a function to impute missing values with the mean. First, there is an if statement to determine whether the variable is numeric before imputation. This method will handle that exception.
mean_impute <- function(x)
{
  if(is.numeric(x))
  {
    # Find the mode of x
    mean_of_x <- mean(x, na.rm = TRUE)
    # Replace the missing by the mode
    library(tidyr)
    x <- replace_na(x, mean_of_x) 
  }
  return(x)    
}

mode_impute <- function(x)
{
  if(!is.numeric(x))
  {
    # Find the mode of x
    mode_of_x <- names(sort(-table(x)))[1]
    # Replace the missing by the mode
    library(tidyr)
    x <- replace_na(x, mode_of_x) 
  }
  return(x)    
}

impute_missing <- function(x)
{
  if(sum(is.na(x)) > 0)
  {
    if(!is.numeric(x))
    {
      mode_of_x <- names(sort(-table(x)))[1]
      library(tidyr)
      x <- replace_na(x, mode_of_x) 
    }
    mean_of_x <- mean(x, na.rm = TRUE)
    library(tidyr)
    x <- replace_na(x, mean_of_x) 
  }
  return(x)
}

# Creating the function to create box plots for dataframes with two variables. This done using 'if' statements to handle each case.

viz <- function(d)
{
  if(is.numeric(d[[1]])&(!is.numeric(d[[2]])))
  {
    d %>% ggplot(aes(x = d[[1]], y = d[[2]]))+
      geom_boxplot()+
      labs(x = names(d)[1], y = names(d)[2])
  }
  else if (!is.numeric(d[[1]])&(is.numeric(d[[2]])))
  {
    d %>% ggplot(aes(x = d[[1]], y = d[[2]]))+
      geom_boxplot()+
      labs(x = names(d)[1], y = names(d)[2])
  }
  else 
  {
    print('This function cannot visualize your data.')
  }
}

library(rpart)
model1 <- function(input_data, train_percent=.8)
{
  library(caret)
  set.seed(00000)
  splitIndex <- createDataPartition(input_data$target, p = train_percent, 
                                    list = FALSE)
  df_train <- input_data[ splitIndex,]
  df_test <- input_data[-splitIndex,]
  forest <- train(target~., data=df_train, 
                          method = "rpart2")
  
  # Evaluating testing accuracy with a confusion matrix on the test data.
  pred <- predict(forest, df_test)
  cm <- confusionMatrix(data = pred, reference = df_test$target)
  cm$overall[1]
}
